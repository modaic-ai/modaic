</ParamField>

<ParamField body="annotations" type="PredictionAnnotation[]" required>

  <Expandable title="PredictionAnnotation properties">
    <ParamField body="annotations[].arbiter_repo" type="string" required>
      Arbiter repository name.
    </ParamField>
    <ParamField body="annotations[].ground_truth" type="string | null">
      Annotated expected answer.
    </ParamField>
    <ParamField body="annotations[].ground_reasoning" type="string | null">
      Annotated reasoning.
    </ParamField>
  </Expandable>
</ParamField>
